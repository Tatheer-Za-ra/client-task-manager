  tasks table:
  - id (UUID, primary key)
  - user_id (UUID, foreign key to auth.users)
  - title (TEXT, required)
  - description (TEXT, optional)
  - status (TEXT: pending/in_progress/completed)
  - due_date (TIMESTAMP, optional)
  - created_at (TIMESTAMP, auto-set)
